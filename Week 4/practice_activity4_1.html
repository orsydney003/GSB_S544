course_names = []
terms_offered = []

# Read in the HTML from the URL using the requests library.
catalog_url = "http://catalog.calpoly.edu/collegesandprograms/collegeofsciencemathematics/statistics/#courseinventory"
catalog_response = requests.get(catalog_url, headers=HEADERS)

# Use Beautiful Soup to parse this string into a tree called soup
catalog_soup = BeautifulSoup(catalog_response.text, 'html.parser')

# Each course is represented by a <div> with class=courseblock, so you can find all the courses with soup.find_all("div", {"class": "courseblock"})
course_blocks = catalog_soup.find_all("div", {"class": "courseblock"})

# iterate over all courseblock divs
for courseblock in course_blocks:
    # find the course name and number
    course_title_tag = courseblock.find("p", {"class": "courseblocktitle"})
    if course_title_tag:
        course_name = course_title_tag.get_text(strip=True)
        # Optional: Remove units from course name
        if course_title_tag.find("span", {"class": "courseunits"}):
            course_title_tag.find("span", {"class": "courseunits"}).extract()
            course_name = course_title_tag.get_text(strip=True)

        course_names.append(course_name)

    # find the term typically offered
    # The term typically offered is in <p> tag with class=noindent. However, there are several tags with this class; term typically offered is the first one.
    term_tag = courseblock.find("p", {"class": "noindent"})
    if term_tag:
        terms_offered.append(term_tag.get_text(strip=True))
    else:
        terms_offered.append(None) # Append None if term info is not found

# creating a data frame
df_courses = pd.DataFrame({
    "course_name": course_names,
    "term_typically_offered": terms_offered
})
